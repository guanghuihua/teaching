	\section*{例题：向量与等边三角形中的最值问题}
	
	\subsection*{题目}
	
	如图，$\triangle ABC$ 是等边三角形，边长为 $2$，$P$ 是平面上任意一点。求 $\vec{PA} \cdot (\vec{PB} + \vec{PC})$ 的最小值。
	
	\medskip
	\noindent\textbf{补充：}当 $a$, $b$ 均为正数时，平方平均数 $\geq$ 算术平均数 $\geq$ 几何平均数。
	
	\subsection*{解}
	
	取等边 $\triangle ABC$ 的中点 $O$。
	
	设 $\vec{a} = \vec{OA}$，$\vec{b} = \vec{OB}$，$\vec{c} = \vec{OC}$，$\vec{s} = \vec{OP}$，则 $\vec{a} + \vec{b} + \vec{c} = \vec{0}$。
	
	\begin{align}
		\vec{PA} &= \vec{OA} - \vec{OP} = \vec{a} - \vec{s}, \\[6pt]
		\vec{PB} + \vec{PC} &= (\vec{b} - \vec{s}) + (\vec{c} - \vec{s}) = \vec{b} + \vec{c} - 2\vec{s} = -\vec{a} - 2\vec{s}.
	\end{align}
	
	所以
	\begin{align}
		\vec{PA} \cdot (\vec{PB} + \vec{PC}) &= (\vec{a} - \vec{s}) \cdot (-\vec{a} - 2\vec{s}) \notag \\
		&= 2s^2 - a \cdot s - a^2 \notag \\
		&= 2\!\left(s - \frac{1}{4}a\right)^{\!2} - \frac{9}{8}a^2.
	\end{align}
	
	当 $\vec{OP} = \vec{s} = \dfrac{1}{4}\vec{a}$ 时，上式取最小值 $-\dfrac{9}{8}a^2$。
	
	\medskip
	
	因为等边 $\triangle ABC$ 的边长为 $2$，所以 $|\vec{OA}| = |\vec{a}| = \dfrac{2\sqrt{3}}{3}$。
	
	所以
	\[
	-\frac{9}{8}a^2 = -\frac{9}{8} \times \left(\frac{2\sqrt{3}}{3}\right)^{\!2} = -\frac{3}{2}.
	\]
	
	\medskip
	
	因此，当点 $P$ 满足 $\vec{OP} = \dfrac{1}{4}\vec{OA}$ 时，$\vec{PA} \cdot (\vec{PB} + \vec{PC})$ 取最小值 $\boxed{-\dfrac{3}{2}}$。
	
	\subsection*{解题要点}
	
	\begin{itemize}
		\item 利用等边三角形中心 $O$ 的性质：$\vec{OA} + \vec{OB} + \vec{OC} = \vec{0}$，将三个向量化简为一个变量。
		\item 将 $\vec{PA} \cdot (\vec{PB} + \vec{PC})$ 展开后，视为关于 $s$（即 $\vec{OP}$）的二次函数，通过配方法求最小值。
	\end{itemize}
